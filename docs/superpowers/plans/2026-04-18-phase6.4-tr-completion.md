```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text.RegularExpressions;
using Xunit;

namespace AuraCore.Tests.UI.Avalonia.Localization;

/// <summary>
/// Regression gate: detect hardcoded user-facing string literals in UI XAML.
/// Ratchet design — current-state files are grandfathered; test fails when
/// a NEW file gains offenders or a grandfathered file gets cleaned up (list
/// must be kept in sync with reality).
/// </summary>
public class HardcodedStringScannerTests
{
    // Relative to repo root. Alphabetical.
    // When you localize a module, REMOVE its path from this list.
    // When a new file legitimately needs a hardcoded literal, add a whitelist
    // entry to `IsWhitelistedString` — don't expand this list.
    private static readonly HashSet<string> GrandfatheredFiles = new(StringComparer.OrdinalIgnoreCase)
    {
        // POPULATED AT STEP 2 — leave empty for the first run.
    };

    // XAML attributes that carry user-facing text.
    private static readonly string[] ScannedAttributes =
    {
        "Text", "Content", "ToolTip", "ToolTip.Tip",
        "Watermark", "Header", "Title", "Label",
    };

    // File globs to scan. Exclude developer-only surfaces.
    private static readonly string[] ExcludedFiles =
    {
        "Dev/ComponentGalleryWindow.axaml",
    };

    [Fact]
    public void No_new_files_with_hardcoded_UI_strings()
    {
        var offenderFiles = ScanViewsForOffenders().ToHashSet(StringComparer.OrdinalIgnoreCase);
        var newOffenders = offenderFiles.Except(GrandfatheredFiles).OrderBy(p => p).ToList();
        var stalelyGrandfathered = GrandfatheredFiles.Except(offenderFiles).OrderBy(p => p).ToList();

        var issues = new List<string>();
        if (newOffenders.Count > 0)
        {
            issues.Add($"New files with hardcoded UI strings (add LocalizationService keys + use bindings):");
            issues.AddRange(newOffenders.Select(f => $"  + {f}"));
        }
        if (stalelyGrandfathered.Count > 0)
        {
            issues.Add($"Files cleaned up but still in GrandfatheredFiles (remove them):");
            issues.AddRange(stalelyGrandfathered.Select(f => $"  - {f}"));
        }

        Assert.True(issues.Count == 0, string.Join(Environment.NewLine, issues));
    }

    private static IEnumerable<string> ScanViewsForOffenders()
    {
        var viewsRoot = FindViewsRoot();
        if (viewsRoot == null) yield break;

        var repoRoot = FindRepoRoot(viewsRoot);
        var axamlFiles = Directory.EnumerateFiles(viewsRoot, "*.axaml", SearchOption.AllDirectories);

        foreach (var file in axamlFiles)
        {
            var rel = Path.GetRelativePath(repoRoot, file).Replace('\\', '/');
            if (ExcludedFiles.Any(ex => rel.EndsWith(ex.Replace('\\', '/'), StringComparison.OrdinalIgnoreCase)))
                continue;

            if (FileHasHardcodedOffenders(file))
                yield return rel;
        }
    }

    private static bool FileHasHardcodedOffenders(string path)
    {
        var text = File.ReadAllText(path);
        foreach (var attr in ScannedAttributes)
        {
            // Match e.g. Text="Hello" or Text ="Hello" (whitespace tolerant)
            var pattern = $@"\b{Regex.Escape(attr)}\s*=\s*""([^""]*)""";
            foreach (Match m in Regex.Matches(text, pattern))
            {
                var value = m.Groups[1].Value;
                if (!IsWhitelistedString(value))
                    return true;
            }
        }
        return false;
    }

    private static bool IsWhitelistedString(string value)
    {
        // Binding expressions - always OK.
        if (value.StartsWith("{", StringComparison.Ordinal)) return true;

        // Empty or whitespace-only.
        if (string.IsNullOrWhiteSpace(value)) return true;

        // Pure numeric (e.g. "0", "100", "3.14", "1/3").
        if (value.All(c => char.IsDigit(c) || c == '.' || c == ',' || c == '/' || c == '-' || c == '+' || c == ' '))
            return true;

        var trimmed = value.Trim();

        // Single char (punctuation or letter — too short to be a sentence).
        if (trimmed.Length <= 1) return true;

        // Pure non-alphanumeric (separators, arrows, bullets).
        if (!trimmed.Any(char.IsLetterOrDigit)) return true;

        // All-uppercase acronyms ≤ 5 chars (CPU, RAM, GPU, OS, SMART, API, URL).
        if (trimmed.Length <= 5 && trimmed.All(c => !char.IsLetter(c) || char.IsUpper(c)) && trimmed.Any(char.IsLetter))
            return true;

        return false;
    }

    private static string? FindViewsRoot()
    {
        // Walk up from assembly location to find the repo's src/UI path.
        var dir = new DirectoryInfo(AppContext.BaseDirectory);
        while (dir != null)
        {
            var candidate = Path.Combine(dir.FullName, "src", "UI", "AuraCore.UI.Avalonia", "Views");
            if (Directory.Exists(candidate)) return candidate;
            dir = dir.Parent;
        }
        return null;
    }

    private static string FindRepoRoot(string viewsRoot)
    {
        // viewsRoot = <repoRoot>/src/UI/AuraCore.UI.Avalonia/Views
        return Path.GetFullPath(Path.Combine(viewsRoot, "..", "..", "..", ".."));
    }
}
```

- [ ] **Step 4: Swap XAML literals for bindings**

Mirror the pattern used by already-localized views. Two acceptable patterns — pick whichever the existing `SidebarView.axaml` uses for OTHER localized text (if any). If the file is entirely un-localized so far, prefer:

**Pattern A (code-behind assignment — preferred when code-behind already has logic):**

XAML keeps a bare `x:Name`:
```xml
<TextBlock x:Name="NavDashboardText"/>
```

Code-behind `SidebarView.axaml.cs` in a new `RefreshLocalizedStrings()` method:
```csharp
private void RefreshLocalizedStrings()
{
    NavDashboardText.Text = LocalizationService.Get("sidebar.dashboard");
    NavAiFeaturesText.Text = LocalizationService.Get("sidebar.aiFeatures");
    // ...
}
```

Call from `OnLoaded` + wire `LocalizationService.LanguageChanged += (_, _) => RefreshLocalizedStrings();` in the constructor (with a corresponding `-=` on `Unloaded` to avoid leaks).

**Pattern B (XAML binding — if the file already uses this pattern):**

```xml
<TextBlock Text="{Binding Source={x:Static local:LocalizationService.Instance}, Path=[sidebar.dashboard]}"/>
```

Requires an indexable instance. The existing codebase uses Pattern A more often (per `SettingsView.axaml.cs` in Phase 6.3). Use Pattern A unless the file clearly uses B.

- [ ] **Step 5: Wire `LanguageChanged` if not already wired**

In `SidebarView.axaml.cs`:

```csharp
public SidebarView()
{
    InitializeComponent();
    Loaded += OnLoaded;
    Unloaded += OnUnloaded;
}

private void OnLoaded(object? sender, RoutedEventArgs e)
{
    RefreshLocalizedStrings();
    LocalizationService.LanguageChanged += OnLanguageChanged;
}

private void OnUnloaded(object? sender, RoutedEventArgs e)
{
    LocalizationService.LanguageChanged -= OnLanguageChanged;
}

private void OnLanguageChanged() => RefreshLocalizedStrings();
```

Adapt to whatever lifecycle pattern the file already has (many views have their own `Loaded`/`Unloaded` handlers). Don't introduce a conflict.
