```csharp
using System.IO;
using System.Text.RegularExpressions;

namespace AuraCore.PrivilegedService.Ops;

public static class ArgumentValidator
{
    private static readonly char[] ShellMetaChars = { ';', '|', '&', '`', '$', '>', '<', '\n', '\r', '\0' };
    private static readonly Regex ServiceNameRegex = new(@"^[A-Za-z0-9_\-\.]{1,256}$", RegexOptions.Compiled);
    private static readonly Regex SubstitutionRegex = new(@"\$\(|\$\{", RegexOptions.Compiled);

    public static bool IsSafeArgument(string? value)
    {
        if (string.IsNullOrEmpty(value)) return false;
        if (value.IndexOfAny(ShellMetaChars) >= 0) return false;
        if (SubstitutionRegex.IsMatch(value)) return false;
        return true;
    }

    public static bool IsPathUnderPrefix(string path, string prefix)
    {
        if (string.IsNullOrWhiteSpace(path) || string.IsNullOrWhiteSpace(prefix)) return false;

        try
        {
            var full = Path.GetFullPath(path);
            var fullPrefix = Path.GetFullPath(prefix);
            if (!fullPrefix.EndsWith(Path.DirectorySeparatorChar))
                fullPrefix += Path.DirectorySeparatorChar;
            return full.StartsWith(fullPrefix, System.StringComparison.OrdinalIgnoreCase);
        }
        catch
        {
            return false;
        }
    }

    public static bool IsValidServiceName(string? name)
        => !string.IsNullOrEmpty(name) && ServiceNameRegex.IsMatch(name);
}
```

```csharp
using System;
using System.Diagnostics;
using System.IO;
using System.Threading;
using System.Threading.Tasks;
using AuraCore.IPC.Contracts;

namespace AuraCore.PrivilegedService.Ops;

public sealed class DriverOperations
{
    private const string DriverBackupPrefix = @"C:\ProgramData\AuraCorePro\DriverBackup\";

    public async Task<DriverOperationResponse> ScanAsync(CancellationToken ct = default)
    {
        return await RunProcessAsync("pnputil.exe", "/scan-devices", ct);
    }

    public async Task<DriverOperationResponse> ExportAsync(string backupDirectory, CancellationToken ct = default)
    {
        if (!ArgumentValidator.IsSafeArgument(backupDirectory))
            return new DriverOperationResponse(false, "", -1, "invalid backup path");
        if (!ArgumentValidator.IsPathUnderPrefix(backupDirectory, DriverBackupPrefix))
            return new DriverOperationResponse(false, "", -1, "path outside allowed prefix");

        Directory.CreateDirectory(backupDirectory);
        return await RunProcessAsync("pnputil.exe", $"/export-driver * \"{backupDirectory}\"", ct);
    }

    private static async Task<DriverOperationResponse> RunProcessAsync(string exe, string args, CancellationToken ct)
    {
        var psi = new ProcessStartInfo(exe, args)
        {
            RedirectStandardOutput = true,
            RedirectStandardError = true,
            UseShellExecute = false,
            CreateNoWindow = true
        };

        using var proc = Process.Start(psi)!;
        var stdoutTask = proc.StandardOutput.ReadToEndAsync();
        var stderrTask = proc.StandardError.ReadToEndAsync();

        using var linkedCts = CancellationTokenSource.CreateLinkedTokenSource(ct);
        linkedCts.CancelAfter(TimeSpan.FromSeconds(60));

        try
        {
            await proc.WaitForExitAsync(linkedCts.Token);
        }
        catch (OperationCanceledException)
        {
            try { proc.Kill(entireProcessTree: true); } catch { }
            return new DriverOperationResponse(false, "", -2, "timeout");
        }

        var stdout = await stdoutTask;
        var stderr = await stderrTask;
        return new DriverOperationResponse(proc.ExitCode == 0, stdout, proc.ExitCode, proc.ExitCode == 0 ? null : stderr);
    }
}
```

```csharp
using System;
using System.Diagnostics;
using System.Threading;
using System.Threading.Tasks;
using AuraCore.IPC.Contracts;

namespace AuraCore.PrivilegedService.Ops;

public sealed class DefenderOperations
{
    public async Task<DefenderActionResponse> ExecuteAsync(DefenderAction action, string? target, CancellationToken ct = default)
    {
        var (cmd, requiresTarget) = BuildPowerShellCommand(action, target);
        if (cmd is null)
            return new DefenderActionResponse(false, "", -1, "invalid action parameters");

        return await RunPowerShellAsync(cmd, ct);
    }

    private static (string? cmd, bool requiresTarget) BuildPowerShellCommand(DefenderAction action, string? target)
    {
        switch (action)
        {
            case DefenderAction.UpdateSignatures:
                return ("Update-MpSignature -ErrorAction Stop", false);

            case DefenderAction.QuickScan:
                return ("Start-MpScan -ScanType QuickScan -ErrorAction Stop", false);

            case DefenderAction.FullScan:
                return ("Start-MpScan -ScanType FullScan -ErrorAction Stop", false);

            case DefenderAction.SetRealtimeEnabled:
                return ("Set-MpPreference -DisableRealtimeMonitoring $false -ErrorAction Stop", false);

            case DefenderAction.SetRealtimeDisabled:
                return ("Set-MpPreference -DisableRealtimeMonitoring $true -ErrorAction Stop", false);

            case DefenderAction.AddExclusion:
                if (string.IsNullOrEmpty(target) || !ArgumentValidator.IsSafeArgument(target))
                    return (null, true);
                return ($"Add-MpPreference -ExclusionPath '{target.Replace("'", "''")}' -ErrorAction Stop", true);

            case DefenderAction.RemoveExclusion:
                if (string.IsNullOrEmpty(target) || !ArgumentValidator.IsSafeArgument(target))
                    return (null, true);
                return ($"Remove-MpPreference -ExclusionPath '{target.Replace("'", "''")}' -ErrorAction Stop", true);

            case DefenderAction.RemoveThreat:
                if (string.IsNullOrEmpty(target) || !long.TryParse(target, out _))
                    return (null, true);
                return ($"Remove-MpThreat -ThreatID {target} -ErrorAction Stop", true);

            default:
                return (null, false);
        }
    }

    private static async Task<DefenderActionResponse> RunPowerShellAsync(string command, CancellationToken ct)
    {
        var psi = new ProcessStartInfo("powershell.exe", $"-NoProfile -NonInteractive -Command \"{command.Replace("\"", "`\"")}\"")
        {
            RedirectStandardOutput = true,
            RedirectStandardError = true,
            UseShellExecute = false,
            CreateNoWindow = true
        };

        using var proc = Process.Start(psi)!;
        var stdoutTask = proc.StandardOutput.ReadToEndAsync();
        var stderrTask = proc.StandardError.ReadToEndAsync();

        using var linkedCts = CancellationTokenSource.CreateLinkedTokenSource(ct);
        linkedCts.CancelAfter(TimeSpan.FromMinutes(2));

        try
        {
            await proc.WaitForExitAsync(linkedCts.Token);
        }
```

```csharp
using System;
using System.Diagnostics;
using System.Threading;
using System.Threading.Tasks;
using AuraCore.IPC.Contracts;

namespace AuraCore.PrivilegedService.Ops;

public sealed class ServiceOperations
{
    public async Task<ServiceControlResponse> ExecuteAsync(string serviceName, ServiceAction action, CancellationToken ct = default)
    {
        if (!ArgumentValidator.IsValidServiceName(serviceName))
            return new ServiceControlResponse(false, "", -1, "invalid service name");

        return action switch
        {
            ServiceAction.Start                 => await RunAsync($"start {serviceName}", ct),
            ServiceAction.Stop                  => await RunAsync($"stop {serviceName}", ct),
            ServiceAction.Restart               => await RestartAsync(serviceName, ct),
            ServiceAction.SetStartupAutomatic   => await RunAsync($"config {serviceName} start= auto", ct),
            ServiceAction.SetStartupManual      => await RunAsync($"config {serviceName} start= demand", ct),
            ServiceAction.SetStartupDisabled    => await RunAsync($"config {serviceName} start= disabled", ct),
            _ => new ServiceControlResponse(false, "", -1, "unknown action")
        };
    }

    private static async Task<ServiceControlResponse> RestartAsync(string name, CancellationToken ct)
    {
        var stopResp = await RunAsync($"stop {name}", ct);
        // sc.exe stop returns non-zero if service already stopped; treat 1062 / 1 as non-fatal
        await Task.Delay(1500, ct);
        var startResp = await RunAsync($"start {name}", ct);
        return new ServiceControlResponse(
            startResp.Success,
            $"stop: {stopResp.Output}\nstart: {startResp.Output}",
            startResp.ExitCode,
            startResp.Error);
    }

    private static async Task<ServiceControlResponse> RunAsync(string args, CancellationToken ct)
    {
        var psi = new ProcessStartInfo("sc.exe", args)
        {
            RedirectStandardOutput = true,
            RedirectStandardError = true,
            UseShellExecute = false,
            CreateNoWindow = true
        };

        using var proc = Process.Start(psi)!;
        var stdoutTask = proc.StandardOutput.ReadToEndAsync();
        var stderrTask = proc.StandardError.ReadToEndAsync();

        using var linkedCts = CancellationTokenSource.CreateLinkedTokenSource(ct);
        linkedCts.CancelAfter(TimeSpan.FromSeconds(30));

        try
        {
            await proc.WaitForExitAsync(linkedCts.Token);
        }
        catch (OperationCanceledException)
        {
            try { proc.Kill(entireProcessTree: true); } catch { }
            return new ServiceControlResponse(false, "", -2, "timeout");
        }

        var stdout = await stdoutTask;
        var stderr = await stderrTask;
        return new ServiceControlResponse(proc.ExitCode == 0, stdout, proc.ExitCode, proc.ExitCode == 0 ? null : stderr);
    }
```

```csharp
using System;
using System.IO;
using System.IO.Pipes;
using System.Security.AccessControl;
using System.Security.Principal;
using System.Text.Json;
using System.Threading;
using System.Threading.Tasks;
using AuraCore.IPC.Contracts;
using AuraCore.PrivilegedService.Ops;
using Microsoft.Extensions.Logging;

namespace AuraCore.PrivilegedService.IPC;

public sealed class PipeServer
{
    private readonly string _pipeName;
    private readonly DriverOperations _driverOps;
    private readonly DefenderOperations _defenderOps;
    private readonly ServiceOperations _serviceOps;
    private readonly ILogger<PipeServer> _logger;

    private const int MaxRequestBytes = 64 * 1024;

    public PipeServer(
        string pipeName,
        DriverOperations driverOps,
        DefenderOperations defenderOps,
        ServiceOperations serviceOps,
        ILogger<PipeServer> logger)
    {
        _pipeName = pipeName;
        _driverOps = driverOps;
        _defenderOps = defenderOps;
        _serviceOps = serviceOps;
        _logger = logger;
    }

    public async Task RunAsync(CancellationToken ct)
    {
        while (!ct.IsCancellationRequested)
        {
            NamedPipeServerStream stream;
            try
            {
                stream = CreateServerStream();
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Failed to create pipe server stream");
                await Task.Delay(500, ct);
                continue;
            }

            try
            {
                await stream.WaitForConnectionAsync(ct);
                _ = Task.Run(() => HandleConnectionAsync(stream, ct), ct);
            }
            catch (OperationCanceledException)
            {
                await stream.DisposeAsync();
                break;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "Pipe accept failed");
                await stream.DisposeAsync();
            }
        }
    }

    private NamedPipeServerStream CreateServerStream()
    {
        if (OperatingSystem.IsWindows())
        {
            var security = new PipeSecurity();
            var authenticatedUsers = new SecurityIdentifier(WellKnownSidType.AuthenticatedUserSid, null);
            security.AddAccessRule(new PipeAccessRule(authenticatedUsers,
                PipeAccessRights.ReadWrite | PipeAccessRights.CreateNewInstance,
                AccessControlType.Allow));

            return NamedPipeServerStreamAcl.Create(
                _pipeName, PipeDirection.InOut, NamedPipeServerStream.MaxAllowedServerInstances,
                PipeTransmissionMode.Byte, PipeOptions.Asynchronous, 4096, 4096, security);
        }

        return new NamedPipeServerStream(_pipeName, PipeDirection.InOut,
            NamedPipeServerStream.MaxAllowedServerInstances, PipeTransmissionMode.Byte, PipeOptions.Asynchronous);
    }

    private async Task HandleConnectionAsync(NamedPipeServerStream stream, CancellationToken ct)
    {
        try
        {
            var lengthBytes = new byte[4];
            await stream.ReadExactlyAsync(lengthBytes, ct);
            var length = BitConverter.ToInt32(lengthBytes);
            if (length <= 0 || length > MaxRequestBytes)
            {
                _logger.LogWarning("Rejected oversized request: {Length}", length);
                await WriteErrorAsync(stream, "request too large", ct);
                return;
            }

            var requestBuf = new byte[length];
            await stream.ReadExactlyAsync(requestBuf, ct);

            using var doc = JsonDocument.Parse(requestBuf);
            var actionId = doc.RootElement.TryGetProperty("ActionId", out var a) ? a.GetString() : null;

            if (string.IsNullOrEmpty(actionId) || !ActionWhitelist.Windows.IsAllowed(actionId))
            {
                _logger.LogWarning("Rejected non-whitelisted action: {ActionId}", actionId ?? "(null)");
                await WriteErrorAsync(stream, $"action not in whitelist: {actionId}", ct);
                return;
            }

            _logger.LogInformation("Dispatching action: {ActionId}", actionId);
            var response = await DispatchAsync(actionId, doc.RootElement, ct);
            await WriteResponseAsync(stream, response, ct);
        }
        catch (OperationCanceledException) { }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Connection handler failure");
            try { await WriteErrorAsync(stream, "internal error", ct); } catch { }
        }
        finally
        {
            await stream.DisposeAsync();
        }
    }

    private async Task<IpcResponse> DispatchAsync(string actionId, JsonElement req, CancellationToken ct)
    {
        switch (actionId)
        {
            case "driver.scan":
                return await _driverOps.ScanAsync(ct);

            case "driver.export":
                var backupDir = req.TryGetProperty("BackupDirectory", out var bd) ? bd.GetString() ?? "" : "";
                return await _driverOps.ExportAsync(backupDir, ct);

            case "defender.update-signatures":
            case "defender.scan-quick":
            case "defender.scan-full":
            case "defender.set-realtime":
            case "defender.add-exclusion":
            case "defender.remove-exclusion":
            case "defender.remove-threat":
                var defAction = req.GetProperty("Action").GetInt32();
                var defTarget = req.TryGetProperty("Target", out var t) ? t.GetString() : null;
                return await _defenderOps.ExecuteAsync((DefenderAction)defAction, defTarget, ct);

            case "service.start":
            case "service.stop":
            case "service.restart":
            case "service.set-startup":
                var svcName = req.GetProperty("ServiceName").GetString() ?? "";
                var svcAction = req.GetProperty("Action").GetInt32();
                return await _serviceOps.ExecuteAsync(svcName, (ServiceAction)svcAction, ct);

            default:
                return new IpcResponse(false, $"no dispatcher for {actionId}");
        }
    }

    private static async Task WriteResponseAsync(NamedPipeServerStream stream, IpcResponse resp, CancellationToken ct)
    {
        var payload = JsonSerializer.SerializeToUtf8Bytes<IpcResponse>(resp, resp.GetType());
        await stream.WriteAsync(BitConverter.GetBytes(payload.Length), ct);
        await stream.WriteAsync(payload, ct);
        await stream.FlushAsync(ct);
    }

    private static async Task WriteErrorAsync(NamedPipeServerStream stream, string error, CancellationToken ct)
    {
        await WriteResponseAsync(stream, new IpcResponse(false, error), ct);
    }
}
```

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.ServiceProcess;
using System.Threading;
using System.Threading.Tasks;
using AuraCore.Application.Interfaces.Platform;

namespace AuraCore.Module.ServiceManager;

public sealed record ServiceOperationOutcome(bool Success, string Output, bool HelperMissing, string? Error = null);

public sealed class ServiceManagerEngine
{
    private readonly IShellCommandService _shell;

    public ServiceManagerEngine(IShellCommandService shell) { _shell = shell; }

    public Task<IReadOnlyList<ServiceEntry>> ListServicesAsync(CancellationToken ct = default)
    {
        // Read-only: ServiceController does not need elevation.
        if (!OperatingSystem.IsWindows())
            return Task.FromResult<IReadOnlyList<ServiceEntry>>(Array.Empty<ServiceEntry>());

        var list = ServiceController.GetServices()
            .Select(sc => new ServiceEntry(
                sc.ServiceName,
                sc.DisplayName,
                sc.Status.ToString(),
                sc.StartType.ToString()))
            .OrderBy(s => s.DisplayName)
            .ToArray();
        return Task.FromResult<IReadOnlyList<ServiceEntry>>(list);
    }

    public Task<ServiceOperationOutcome> StartAsync(string name, CancellationToken ct = default)
        => DispatchAsync("service.start", new[] { name }, ct);

    public Task<ServiceOperationOutcome> StopAsync(string name, CancellationToken ct = default)
        => DispatchAsync("service.stop", new[] { name }, ct);

    public Task<ServiceOperationOutcome> RestartAsync(string name, CancellationToken ct = default)
        => DispatchAsync("service.restart", new[] { name }, ct);

    public Task<ServiceOperationOutcome> SetStartupAsync(string name, string mode, CancellationToken ct = default)
    {
        if (mode != "auto" && mode != "demand" && mode != "disabled")
            return Task.FromResult(new ServiceOperationOutcome(false, "", false, $"invalid mode {mode}"));
        return DispatchAsync("service.set-startup", new[] { name, mode }, ct);
    }

    private async Task<ServiceOperationOutcome> DispatchAsync(string id, string[] args, CancellationToken ct)
    {
        var result = await _shell.RunPrivilegedAsync(new PrivilegedCommand(id, "sc.exe", args, 30), ct);
        return new ServiceOperationOutcome(result.Success, result.Stdout,
            result.AuthResult == PrivilegeAuthResult.HelperMissing,
            result.Success ? null : result.Stderr);
    }
}
```

```csharp
public sealed record DirectoryEntry(string Path, long SizeBytes, bool IsDirectory, bool HasChildren);

public partial class SpaceAnalyzerEngine
{
    public Task<IReadOnlyList<DirectoryEntry>> EnumerateChildrenAsync(string path)
    {
        if (!Directory.Exists(path))
            return Task.FromResult<IReadOnlyList<DirectoryEntry>>(System.Array.Empty<DirectoryEntry>());

        var list = new List<DirectoryEntry>();

        foreach (var dir in Directory.EnumerateDirectories(path))
        {
            var hasChildren = false;
            long size = 0;
            try
            {
                size = DirectorySizeSafe(dir);
                hasChildren = Directory.EnumerateFileSystemEntries(dir).Any();
            }
            catch { /* unauthorized / deleted mid-scan: ignore */ }
            list.Add(new DirectoryEntry(dir, size, IsDirectory: true, HasChildren: hasChildren));
        }

        foreach (var file in Directory.EnumerateFiles(path))
        {
            long size = 0;
            try { size = new FileInfo(file).Length; } catch { }
            list.Add(new DirectoryEntry(file, size, IsDirectory: false, HasChildren: false));
        }

        return Task.FromResult<IReadOnlyList<DirectoryEntry>>(list
            .OrderByDescending(e => e.SizeBytes)
            .ToList());
    }

    private static long DirectorySizeSafe(string dir)
    {
        long total = 0;
        try
        {
            foreach (var f in Directory.EnumerateFiles(dir, "*", SearchOption.AllDirectories))
            {
                try { total += new FileInfo(f).Length; } catch { }
            }
        }
        catch { }
        return total;
    }
}
```

```csharp
using System.Collections.ObjectModel;
using System.ComponentModel;
using System.Runtime.CompilerServices;
using System.Threading.Tasks;
using AuraCore.Module.SpaceAnalyzer;

namespace AuraCore.UI.Avalonia.ViewModels.SpaceAnalyzer;

public sealed class DirectoryNodeVM : INotifyPropertyChanged
{
    private readonly SpaceAnalyzerEngine _engine;
    private bool _isExpanded;
    private bool _childrenLoaded;

    public string Path { get; }
    public string DisplayName { get; }
    public long SizeBytes { get; }
    public bool IsDirectory { get; }
    public bool HasChildren { get; }
    public ObservableCollection<DirectoryNodeVM> Children { get; } = new();

    public bool IsExpanded
    {
        get => _isExpanded;
        set
        {
            if (_isExpanded == value) return;
            _isExpanded = value;
            RaisePropertyChanged();
            if (value && !_childrenLoaded) _ = LoadChildrenAsync();
        }
    }

    public DirectoryNodeVM(SpaceAnalyzerEngine engine, DirectoryEntry entry)
    {
        _engine = engine;
        Path = entry.Path;
        DisplayName = System.IO.Path.GetFileName(entry.Path) ?? entry.Path;
        SizeBytes = entry.SizeBytes;
        IsDirectory = entry.IsDirectory;
        HasChildren = entry.HasChildren;

        if (HasChildren) Children.Add(new DirectoryNodeVM(engine) { DisplayName = "(loading…)" });
    }

    private DirectoryNodeVM(SpaceAnalyzerEngine engine)
    {
        _engine = engine;
        Path = "";
        DisplayName = "";
    }

    private async Task LoadChildrenAsync()
    {
        _childrenLoaded = true;
        Children.Clear();
        var entries = await _engine.EnumerateChildrenAsync(Path);
        foreach (var e in entries)
            Children.Add(new DirectoryNodeVM(_engine, e));
    }

    public event PropertyChangedEventHandler? PropertyChanged;
    private void RaisePropertyChanged([CallerMemberName] string? name = null)
        => PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(name));
}
```
